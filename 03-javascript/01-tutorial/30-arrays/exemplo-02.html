Looping Array Elements - Array.forEach() function

Script

        
          const fruits = ["Banana", "Orange", "Apple", "Mango"];
          let txt = `<ul>`;
          fruits.forEach(myFunction);
          txt += `</ul>`;
    
          function myFunction(value) {
            txt += `<li>${value}</li>`
          }
    
          document.getElementById("preview").innerHTML = txt;
